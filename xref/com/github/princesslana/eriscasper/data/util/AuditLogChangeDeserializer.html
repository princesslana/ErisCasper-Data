22    public AuditLogChangeDeserializer() {
23      super((Class<?>) null);

26    @Override
27    public AuditLogChange deserialize(JsonParser json, DeserializationContext context)
28        throws IOException {
29      JsonNode node = json.getCodec().readTree(json);
30      return parseJson(node, context, json.getCodec());
31    }

33    private AuditLogChange parseJson(JsonNode node, DeserializationContext context, ObjectCodec codec)
34        throws JsonProcessingException {
35      String key = node.get("key").textValue();
36      switch (key.toLowerCase()) {
37        case "$add":
38        case "$remove":
39          return newArrayInstance(node, Role.class, key, codec);
40        case "permission_overwrites":
41          return newArrayInstance(node, Overwrite.class, key, codec);
42        case "owner_id":
43        case "afk_channel_id":
44        case "widget_channel_id":
45        case "system_channel_id":
46        case "channel_id":
47        case "inviter_id":
48        case "id":
49        case "application_id":
50          return newInstance(node, Snowflake.class, key, codec);
51        case "widget_enabled":
52        case "nsfw":
53        case "temporary":
54        case "deaf":
55        case "mute":
56        case "hoist":
57        case "mentionable":
58          return newInstance(node, Boolean.class, key, codec);
59        case "permissions":
60        case "color":
61        case "allow":
62        case "deny":
63        case "max_uses":
64        case "uses":
65        case "max_age":
66        case "afk_timeout":
67        case "mfa_level":
68        case "verification_level":
69        case "explicit_content_filter":
70        case "default_message_notifications":
71        case "position":
72        case "bitrate":
73        case "prune_delete_days":
74          return newInstance(node, Integer.class, key, codec);
75        case "type":
76        case "nick":
77        case "avatar_hash":
78        case "name":
79        case "icon_hash":
80        case "splash_hash":
81        case "code":
82        case "region":
83        case "vanity_url_code":
84        case "topic":
85          return newInstance(node, String.class, key, codec);
86        default:
87          // just for stuff not represented by discord (like system_channel_id wasn't)
88          // we will catch them as they're reported, can't predict stuff with no documentation for it
89          throw context.instantiationException(
90              AuditLogChange.class, "Failed to load " + key + " from " + node.asText());
91      }
92    }

94    private <T> AuditLogChange newInstance(
95        JsonNode node, Class<T> clazz, String key, ObjectCodec codec) throws JsonProcessingException {
96      T newValue = null;
97      T oldValue = null;
98      if (node.has("new_value")) {
99        newValue = parse(node.get("new_value"), clazz, codec);
100     }
101     if (node.has("old_value")) {
102       oldValue = parse(node.get("old_value"), clazz, codec);
103     }
104     return ImmutableAuditLogChange.builder()
105         .key(key)
106         .newValue(Optional.ofNullable(newValue))
107         .oldValue(Optional.ofNullable(oldValue))
108         .build();
109   }
110 
111   private <T> AuditLogChange newArrayInstance(
112       JsonNode node, Class<T> clazz, String key, ObjectCodec codec) throws JsonProcessingException {
113     ImmutableList<T> newValue = ImmutableList.of();
114     ImmutableList<T> oldValue = ImmutableList.of();
115     if (node.has("new_value")) {
116       newValue = parseArray(node.get("new_value"), clazz, codec);
117     }
118     if (node.has("old_value")) {
119       oldValue = parseArray(node.get("old_value"), clazz, codec);
120     }
121     return ImmutableAuditLogChange.builder().key(key).newValue(newValue).oldValue(oldValue).build();
122   }
123 
124   private <T> ImmutableList<T> parseArray(JsonNode node, Class<T> clazz, ObjectCodec codec)
125       throws JsonProcessingException {
126     ArrayNode array = (ArrayNode) node;
127     ImmutableList.Builder<T> builder = ImmutableList.builder();
128     Iterator<JsonNode> arrayTree = array.elements();
129     while (arrayTree.hasNext()) {
130       builder.add(parse(arrayTree.next(), clazz, codec));
131     }
132     return builder.build();
133   }
134 
135   private <T> T parse(JsonNode node, Class<T> clazz, ObjectCodec codec)
136       throws JsonProcessingException {
137     return codec.treeToValue(node, clazz);
138   }
