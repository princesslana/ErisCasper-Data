8   public class TestUserWithGuildId {
9     @Test
10    public void fromJson_whenValidPayload_shouldDeserialize() {
11      String payload =
12          "{     \"username\"     :\"LaBotuel\","
13              + "\"id\"           :\"417388135027048495\","
14              + "\"discriminator\":\"7013\","
15              + "\"bot\"          :true,"
16              + "\"avatar\"       :null,"
17              + "\"guild_id\"     :\"123456789123456789\""
18              + "}";
19  
20      DataAssert.thatFromJson(payload, UserWithGuildId.class)
21          .hasFieldOrPropertyWithValue("guildId", Snowflake.of("123456789123456789"))
22          .hasFieldOrPropertyWithValue("user.id", Snowflake.of("417388135027048495"))
23          .hasFieldOrPropertyWithValue("user.username", "LaBotuel")
24          .hasFieldOrPropertyWithValue("user.discriminator", "7013")
25          .hasFieldOrPropertyWithValue("user.avatar", Optional.empty())
26          .hasFieldOrPropertyWithValue("user.bot", Optional.of(true))
27          .hasFieldOrPropertyWithValue("user.mfaEnabled", Optional.empty())
28          .hasFieldOrPropertyWithValue("user.verified", Optional.empty())
29          .hasFieldOrPropertyWithValue("user.email", Optional.empty());
30    }
