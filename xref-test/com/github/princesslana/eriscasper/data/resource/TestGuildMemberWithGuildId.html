9   public class TestGuildMemberWithGuildId {
10    @Test
11    public void fromJson_whenValidPayload_shouldDeserialize() {
12      String payload =
13          "{"
14              + "\"user\":"
15              + "{"
16              + "   \"username\"     :\"LaBotuel\","
17              + "   \"id\"           :\"417388135027048495\","
18              + "   \"discriminator\":\"7013\","
19              + "   \"bot\"          :true,"
20              + "   \"avatar\"       :null"
21              + "},"
22              + "\"joined_at\":\"2018-06-03T10:59:19.884195-04:00\","
23              + "\"deaf\":false,"
24              + "\"mute\":false,"
25              + "\"guild_id\"     :\"123456789123456789\""
26              + "}";
27  
28      DataAssert.thatFromJson(payload, GuildMemberWithGuildId.class)
29          .hasFieldOrPropertyWithValue("guildId", Snowflake.of("123456789123456789"))
30          .hasFieldOrPropertyWithValue("member.user.id", Snowflake.of("417388135027048495"))
31          .hasFieldOrPropertyWithValue("member.deaf", false)
32          .hasFieldOrPropertyWithValue("member.mute", false)
33          .hasFieldOrPropertyWithValue("member.user.username", "LaBotuel")
34          .hasFieldOrPropertyWithValue("member.user.discriminator", "7013")
35          .hasFieldOrPropertyWithValue("member.user.avatar", Nullable.ofNull())
36          .hasFieldOrPropertyWithValue("member.user.bot", Optional.of(true))
37          .hasFieldOrPropertyWithValue("member.user.mfaEnabled", Optional.empty())
38          .hasFieldOrPropertyWithValue("member.user.verified", Optional.empty())
39          .hasFieldOrPropertyWithValue("member.user.email", Optional.empty());
40    }
