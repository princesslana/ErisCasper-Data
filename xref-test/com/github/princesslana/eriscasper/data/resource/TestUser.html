3   import com.github.princesslana.eriscasper.data.DataAssert;
4   import com.github.princesslana.eriscasper.data.Snowflake;
5   import com.github.princesslana.eriscasper.data.util.Nullable;
6   import java.util.Optional;
7   import org.testng.annotations.Test;
8   
9   public class TestUser {
10  
11    @Test
12    public void fromJson_whenValidPayload_shouldDeserialize() {
13      String payload =
14          "{     \"username\"     :\"LaBotuel\","
15              + "\"id\"           :\"417388135027048495\","
16              + "\"discriminator\":\"7013\","
17              + "\"bot\"          :true,"
18              + "\"avatar\"       :null"
19              + "}";
20  
21      DataAssert.thatFromJson(payload, User.class)
22          .hasFieldOrPropertyWithValue("id", Snowflake.of("417388135027048495"))
23          .hasFieldOrPropertyWithValue("username", "LaBotuel")
24          .hasFieldOrPropertyWithValue("discriminator", "7013")
25          .hasFieldOrPropertyWithValue("avatar", Nullable.ofNull())
26          .hasFieldOrPropertyWithValue("bot", Optional.of(true))
27          .hasFieldOrPropertyWithValue("mfaEnabled", Optional.empty())
28          .hasFieldOrPropertyWithValue("verified", Optional.empty())
29          .hasFieldOrPropertyWithValue("email", Optional.empty());
30    }
31  
32    /**
33     * @see <a href="https://discordapp.com/developers/docs/resources/user#user-object-example-user">
34     *     https://discordapp.com/developers/docs/resources/user#user-object-example-user</a>
35     */
36    @Test
37    public void fromJson_whenExamplePayload_shouldDeseralize() {
38      String payload =
39          "{     \"id\"           : \"80351110224678912\","
40              + "\"username\"     : \"Nelly\","
41              + "\"discriminator\": \"1337\","
42              + "\"avatar\"       : \"8342729096ea3675442027381ff50dfe\","
43              + "\"verified\"     : true,"
44              + "\"email\"        : \"nelly@discordapp.com\""
45              + "}";
46  
47      DataAssert.thatFromJson(payload, User.class)
48          .hasFieldOrPropertyWithValue("id", Snowflake.of("80351110224678912"))
49          .hasFieldOrPropertyWithValue("username", "Nelly")
50          .hasFieldOrPropertyWithValue("discriminator", "1337")
51          .hasFieldOrPropertyWithValue("avatar", Nullable.of("8342729096ea3675442027381ff50dfe"))
52          .hasFieldOrPropertyWithValue("bot", Optional.empty())
53          .hasFieldOrPropertyWithValue("mfaEnabled", Optional.empty())
54          .hasFieldOrPropertyWithValue("verified", Optional.of(true))
55          .hasFieldOrPropertyWithValue("email", Optional.of("nelly@discordapp.com"));
56    }
