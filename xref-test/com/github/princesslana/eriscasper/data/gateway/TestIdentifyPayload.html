3   import com.github.princesslana.eriscasper.data.Data;
4   import com.github.princesslana.eriscasper.data.DataException;
5   import java.util.Map;
6   import java.util.Optional;
7   import org.assertj.core.api.Assertions;
8   import org.testng.annotations.Test;
9   
10  public class TestIdentifyPayload {
11  
12    @Test
13    public void toJsonString_whenAbsentFields_shouldNotIncludeAbsentFields() throws DataException {
14      IdentifyPayload payload =
15          ImmutableIdentifyPayload.builder().token("abc").properties(connectionProperties()).build();
16  
17      Assertions.assertThat(Data.fromJson(Data.toJsonString(payload), Map.class))
18          .hasSize(2)
19          .containsEntry("token", "abc")
20          .containsKey("properties")
21          .doesNotContainKeys("compress", "large_threshold", "shard", "presence");
22    }
23  
24    @Test
25    public void toJsonString_whenSharding_shouldIncludeShardingAsArray() throws DataException {
26      IdentifyPayload payload =
27          ImmutableIdentifyPayload.builder()
28              .token("abc")
29              .properties(connectionProperties())
30              .shard(ShardPayload.of(2, 3))
31              .build();
32  
33      Assertions.assertThat(Data.fromJson(Data.toJsonString(payload), IdentifyPayload.class))
34          .hasFieldOrPropertyWithValue("token", "abc")
35          .hasFieldOrPropertyWithValue("shard", Optional.of(ShardPayload.of(2, 3)));
36    }
37  
38    private static ConnectionPropertiesPayload connectionProperties() {
39      return ImmutableConnectionPropertiesPayload.builder()
40          .os(System.getProperty("os.name"))
41          .browser("ErisCasper-Data")
42          .device("ErisCasper-Data")
43          .build();
44    }
45  }
